    <script>
    document.getElementById('apiCallBtn').addEventListener('click', function() {
        fritos.remoteCall('https://serene-island-81305.herokuapp.com/api/200', {
            method: 'GET',
            timeout: 45,
            headers: {
                'Accept': 'application/json',
            },
            onSuccess: function(data) {
                console.log('GET Success:', data);
              
            },
            onError: function(err) {
                console.error('GET Error:', err);
                
            }
        });

        fritos.remoteCall('https://serene-island-81305.herokuapp.com/api/201', {
            method: 'POST',
            timeout: 45,
            headers: {
                'Accept': 'application/json',
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({
                id: 1,
                message: 'Testing POST'
            }),
            onSuccess: function(data) {
                console.log('POST Success:', data);
     
            },
            onError: function(err) {
                console.error('POST Error:', err);
     
            }
        });
    });
    </script>
